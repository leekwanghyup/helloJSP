```JAVA
// 숫자만
^[0-9]*$

// 한글
^[가-힣]*$

// 영어 숫자 
^[a-zA-Z0-9]*$

// 비밀번호 : 숫자문자 특수문자 포함 8 ~15 이내
^.*(?=^.{8,15}$)(?=.*\d)(?=.*[a-zA-Z])(?=.*[!@#$%^&+=]).*$

// 비밀번호 : 숫자 영문자 특수문자를 반드시 포함해야한다.
^.*(?=^.*$)(?=.*\\d)(?=.*[a-zA-Z])(?=.*[!@#$%^&+=]).*$

// 비밀번호 (숫자, 문자 포함의 6~12자리 이내)
^[A-Za-z0-9]{6,12}$

// 이메일
^[0-9a-zA-Z]([-_.]?[0-9a-zA-Z])*@[0-9a-zA-Z]([-_.]?[0-9a-zA-Z])*.[a-zA-Z]{2,3}$

// 일반전화
^\\d{3}-\\d{3,4}-\\d{4}$

// 주민등록번호
\d{6} \- [1-4]\d{6}

// 파일확장자 : '|'를 이용한 OR 연산 txt또는 pdf 또는 xls파일만 허용
^\\S+.(?i)(txt|pdf|hwp|xls)$

// 이중확장자
(.+?)((\\.tar)?\\.gz)$

```
